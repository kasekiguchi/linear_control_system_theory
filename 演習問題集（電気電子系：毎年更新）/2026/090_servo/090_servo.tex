\noindent
{\bf サーボ系（演習）}　次のシステムに対して，一定目標値$r$に対して定常偏差なく追従するコントローラを設計せよ．ただし，閉ループ系のすべての極を$\chrp$とせよ．
\begin{align*}
	\frac{dx}{dt} &= \chrA \cdot x + \chrB \cdot u, \hspace{0cm} &A=\chrA,\  B=\chrB
\\
	y &= \chrC \cdot x , &C=\chrC, \ D=0
\end{align*}

積分器
\begin{align*}
	\frac{d \eta}{dt}&= \ e \ = \ r-y \ = \ r- C x \ = \ r-x
\end{align*}

拡大系
\begin{align*}
	\frac{d}{dt}
	\state	
	& = 
	\left (
	\begin{array}{cc}
	 A & 0 \\
	 -C& 0
	\end{array}
	\right )
	\state
	+
	\left (
	\begin{array}{c}
	 B \\ 0 
	\end{array}
	\right )
	u
	+
	\left (
	\begin{array}{c}
	 0 \\ I 
	\end{array}
	\right )
	r
\\
	\frac{d}{dt}
	\state
	& = 
	\chreA	\state
	+
	\chreB	u
	+
	\left (
	\begin{array}{c}
	 0 \\ 1 
	\end{array}
	\right )
	r
\end{align*}

状態フィードバック
\begin{align*}
	u = \left ( \begin{array}{cc}
			F & K \end{array} \right )
		\state
\end{align*}

閉ループ系
\begin{align*}
	\frac{d}{dt}
	\left (
	\begin{array}{c}
	 x \\ \eta 
	\end{array}
	\right )
	& = 
	\left (
	\begin{array}{cc}
	 A & 0 \\
	 -C& 0
	\end{array}
	\right )
	\state
	+
	\left (
	\begin{array}{c}
	 B \\ 0 
	\end{array}
	\right )
	\left ( \begin{array}{cc}
			F & K \end{array} \right )
	\state
	+
	\left (
	\begin{array}{c}
	 0 \\ I 
	\end{array}
	\right )
	r
\\
	& = 
	\left (
	\begin{array}{cc}
	 A & 0 \\
	 -C& 0
	\end{array}
	\right )
	\state
	+
	\left (
	\begin{array}{cc}
	 B F & BK \\ 0 &0
	\end{array}
	\right )
	\state
	+
	\left (
	\begin{array}{c}
	 0 \\ I 
	\end{array}
	\right )
	r
\\
	& = 
	\left (
	\begin{array}{cc}
	 A+BF & BK \\
	 -C& 0
	\end{array}
	\right )
	\state
	+
	\left (
	\begin{array}{c}
	 0 \\ I 
	\end{array}
	\right )
	r
\\
	\frac{d}{dt}
	\state
	& = 
	\left (
	\begin{array}{cc}
	 \chrABF & \chrBK \\
	 -\chrC & 0
	\end{array}
	\right )
	\state
	+
	\left (
	\begin{array}{c}
	 0 \\ 1 
	\end{array}
	\right )
	r
\end{align*}

閉ループ系の特性多項式
\begin{align*}
	\det
	\left \{
		sI- 
		\left (
		\begin{array}{cc}
		 \chrABF & \chrBK \\
		 -\chrC & 0
		\end{array}
		\right )
	\right \}
	&=
	\det
	\left ( 
	\begin{array}{cc}
	 s-(\chrABF) & -\chrBK \\
	 \chrC & s
	\end{array}
	\right )
\\
	&=
	\{ s-(\chrABF) \} \cdot s + \chrBK * \chrC
	=
	s^2 -(\chrABF)+\chrBK*\chrC
\end{align*}

これは$s$に関する二次方程式なので根（極）は２個.
$\therefore$すべての極が$\chrp$となる特性多項式は
\begin{align*}
	(s+\chrmp)(s+\chrmp) = s^2+ \chrcoefbd s+\chrcoefcd
\end{align*}

係数を比較して
\begin{align*}
	-(\chrABF)=\chrcoefbd, \ \chrBK*\chrC = \chrcoefcd \ \ \Rightarrow \ \ F=\chrsF, \ K=\chrsK
\end{align*}

コントローラーは
\begin{align*}
	\frac{d \eta}{dt}&= \ r-y 
\\
	u &= \left ( \begin{array}{cc}
			\chrsF & \chrsK \end{array} \right )
		\left (
		\begin{array}{c}
		 x \\ \eta 
		\end{array}
		\right )
\end{align*}
